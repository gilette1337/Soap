def check_and_install(package: str, import_name: str = None) -> bool:
    """Check if package is installed, if not, try to install it"""
    if import_name is None:
        import_name = package.replace('-', '_')
    
    try:
        importlib.import_module(import_name)
        return True
    except ImportError:
        print(f"[!] {package} not found, attempting to install...")
        
        # Try with pip3 first
        install_cmds = [
            [sys.executable, '-m', 'pip', 'install', package],
            ['pip3', 'install', package],
            ['pip', 'install', package]
        ]
        
        for cmd in install_cmds:
            try:
                subprocess.run(cmd, check=True, capture_output=True)
                print(f"[+] Successfully installed {package}")
                return True
            except (subprocess.CalledProcessError, FileNotFoundError):
                continue
        
        print(f"[!] Failed to install {package}. Please install manually: pip3 install {package}")
        return False

def check_sudo() -> bool:
    """Check if script is running with sudo"""
    return os.geteuid() == 0 if hasattr(os, 'geteuid') else False

class FastProxyValidator:
    """Phase 1: Ultra-fast proxy validation"""
    
    def __init__(self, proxy_file: str, test_url: str = "http://httpbin.org/ip"):
        self.proxy_file = proxy_file
        self.test_url = test_url
        self.proxies = []
        self.working_proxies = []
    
    def load_proxies(self) -> bool:
        if not os.path.exists(self.proxy_file):
            print(f"{Fore.RED}[!] Proxy file not found: {self.proxy_file}")
            return False
        
        with open(self.proxy_file, 'r') as f:
            raw = [line.strip() for line in f if line.strip() and not line.startswith('#')]
        
        for proxy in raw:
            if '://' not in proxy:
                proxy = f'http://{proxy}'
            if ':' in proxy.split('//')[-1]:
                self.proxies.append(proxy)
        
        print(f"{Fore.GREEN}[+] Loaded {len(self.proxies)} proxies")
        return len(self.proxies) > 0
    
    async def check_single(self, proxy: str, semaphore: asyncio.Semaphore, idx: int, total: int) -> Optional[str]:
        async with semaphore:
            connector = aiohttp.TCPConnector(ssl=False, force_close=True, enable_cleanup_closed=True)
            timeout = aiohttp.ClientTimeout(total=8, connect=5)
            
            try:
                async with aiohttp.ClientSession(connector=connector, timeout=timeout) as session:
                    async with session.get(self.test_url, proxy=proxy) as resp:
                        if resp.status == 200:
                            print(f"\r{Fore.GREEN}[✓] Validating: {idx}/{total} | Working: {len(self.working_proxies)} | {proxy[:40]}...", end="", flush=True)
                            return proxy
                        else:
                            print(f"\r{Fore.YELLOW}[?] Validating: {idx}/{total} | Working: {len(self.working_proxies)} | {proxy[:40]}...", end="", flush=True)
                            return None
            except Exception:
                print(f"\r{Fore.RED}[✗] Validating: {idx}/{total} | Working: {len(self.working_proxies)} | {proxy[:40]}...", end="", flush=True)
                return None
            finally:
                await connector.close()
    
    async def validate_all(self, concurrency: int = 300) -> List[str]:
        print(f"\n{Fore.CYAN}{'='*55}")
        print(f"{Fore.CYAN}📡 PHASE 1: FAST PROXY VALIDATION")
        print(f"{Fore.CYAN}{'='*55}")
        print(f"{Fore.YELLOW}⚡ Concurrency: {concurrency} workers\n")
        
        semaphore = asyncio.Semaphore(concurrency)
        tasks = []
        total = len(self.proxies)
        
        for idx, proxy in enumerate(self.proxies, 1):
            tasks.append(self.check_single(proxy, semaphore, idx, total))
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        self.working_proxies = []
        for r in results:
            if r and isinstance(r, str):
                self.working_proxies.append(r)
        
        print(f"\n\n{Fore.GREEN}{'='*55}")
        print(f"{Fore.GREEN}✅ VALIDATION COMPLETE")
        print(f"{Fore.GREEN}   Working: {len(self.working_proxies)}/{len(self.proxies)}")
        print(f"{Fore.GREEN}   Success Rate: {len(self.working_proxies)/len(self.proxies)*100:.1f}%")
        print(f"{Fore.GREEN}{'='*55}")
        
        if self.working_proxies:
            with open("working_proxies.txt", "w") as f:
                for p in self.working_proxies:
                    f.write(f"{p}\n")
            print(f"{Fore.GREEN}   Saved to: working_proxies.txt")
        
        return self.working_proxies
